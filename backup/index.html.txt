    <script>
      const CONFIG_URL = window.ENV_SUPABASE_URL || '';
      const CONFIG_KEY = window.ENV_SUPABASE_ANON_KEY || '';

      let adminSupabase = null;
      let appRoleState = 'assistant';
      let editingItemId = null;

      document.addEventListener('DOMContentLoaded', () => {
        if (
          window.supabase &&
          typeof window.supabase.createClient === 'function'
        ) {
          adminSupabase = window.supabase.createClient(CONFIG_URL, CONFIG_KEY);
        }

        if (localStorage.getItem('thrift_demo_role')) {
          appRoleState = localStorage.getItem('thrift_demo_role');
        } else {
          localStorage.setItem('thrift_demo_role', appRoleState);
        }

        evaluateRoleLockoutPatterns();
        loadWorkspacePipeline();
        initializeRealtimePipeline();
      });

      function initializeRealtimePipeline() {
        if (!adminSupabase) return;
        adminSupabase
          .channel('schema-db-changes')
          .on(
            'postgres_changes',
            { event: '*', schema: 'public', table: 'items' },
            (payload) => {
              loadWorkspacePipeline();
            }
          )
          .subscribe();
      }

      function evaluateRoleLockoutPatterns() {
        const badge = document.getElementById('role-badge');
        const warningBox = document.getElementById('payout-lock-warning');
        const saveWrapper = document.getElementById('save-button-wrapper');
        const inputs = [
          document.getElementById('bank-name-field'),
          document.getElementById('bank-account-field'),
          document.getElementById('bank-holder-field'),
        ];

        if (appRoleState === 'owner') {
          badge.className =
            'bg-emerald-50 text-emerald-700 text-[10px] font-bold uppercase tracking-wider px-2 py-0.5 rounded-md border border-emerald-200 flex items-center gap-1';
          badge.innerHTML = '<span>👑 Owner Mode</span>';
          if (warningBox) warningBox.classList.add('hidden');
          if (saveWrapper) saveWrapper.classList.remove('hidden');
          inputs.forEach(
            (i) =>
              (i && i.removeAttribute('disabled')) ||
              i?.classList.remove(
                'bg-slate-50',
                'text-slate-400',
                'cursor-not-allowed'
              )
          );
        } else {
          badge.className =
            'bg-amber-50 text-amber-700 text-[10px] font-bold uppercase tracking-wider px-2 py-0.5 rounded-md border border-amber-200 flex items-center gap-1';
          badge.innerHTML = '<span>🛡️ Assistant Profile</span>';
          if (warningBox) warningBox.classList.remove('hidden');
          if (saveWrapper) saveWrapper.classList.add('hidden');
          inputs.forEach(
            (i) =>
              (i && i.setAttribute('disabled', true)) ||
              i?.classList.add(
                'bg-slate-50',
                'text-slate-400',
                'cursor-not-allowed'
              )
          );
        }
      }

      function simulateRoleToggle() {
        appRoleState = appRoleState === 'owner' ? 'assistant' : 'owner';
        localStorage.setItem('thrift_demo_role', appRoleState);
        evaluateRoleLockoutPatterns();
      }

      function switchTab(targetTabId) {
        document
          .querySelectorAll('.tab-content')
          .forEach((el) => el.classList.add('hidden'));
        document
          .getElementById(`tab-content-${targetTabId}`)
          .classList.remove('hidden');

        document.querySelectorAll('.tab-btn').forEach((btn) => {
          btn.classList.remove(
            'bg-white',
            'text-slate-900',
            'shadow-sm',
            'font-bold'
          );
          btn.classList.add('text-slate-500', 'font-semibold');
        });
        const activeBtn = document.getElementById(`tab-btn-${targetTabId}`);
        if (activeBtn)
          activeBtn.classList.add(
            'bg-white',
            'text-slate-900',
            'shadow-sm',
            'font-bold'
          );

        const mobileTabIds = ['dashboard', 'add-drop', 'inventory', 'settings'];
        mobileTabIds.forEach((id) => {
          const mBtn = document.getElementById(`mobile-tab-${id}`);
          if (mBtn) {
            if (id === targetTabId) {
              mBtn.className =
                'flex flex-col items-center gap-1 text-indigo-600 font-bold text-[10px] transition-all';
            } else {
              mBtn.className =
                'flex flex-col items-center gap-1 text-slate-400 font-medium text-[10px] transition-all';
            }
          }
        });
      }

      /* AI IMPLEMENTATION ENGINE BLOCK 1: COPYWRITER GENERATOR */
      function generateAICopywriterCaption() {
        const title = document.getElementById('drop-title-input').value.trim();
        const price = document.getElementById('drop-price-input').value.trim();

        if (!title || !price) {
          alert(
            'Please type a Title and Price first so the AI can build your marketing copy!'
          );
          return;
        }

        const expressions = [
          'Vintage gems only ✨',
          'Steal of the week 🔥',
          'Clean look, premium weight 💎',
          'Hard to find, impossible to forget 🪐',
        ];
        const hooks = [
          'Snooze you lose on this one.',
          'First to pay gets the drop.',
          'No restocks, when it is gone it is truly gone.',
        ];
        const selectedExpression =
          expressions[Math.floor(Math.random() * expressions.length)];
        const selectedHook = hooks[Math.floor(Math.random() * hooks.length)];

        const aiCaptionText = `⚡️ FRESH THRIFT DROP DETECTED ⚡️

🛒 Item: ${title}
💰 Price: ₦${Number(price).toLocaleString()}

${selectedExpression} Cond 10/10. Ready to anchor your daily rotation. 

📌 HOW TO CLAIM:
Click the link in our bio to grab it instantly on the shop floor before the timer locks out! ${selectedHook}

#ThriftTrack #VintageFinds #StreetwearNigeria #SustainableFashion`;

        document.getElementById('ai-copywriter-text').innerText = aiCaptionText;
        document.getElementById('ai-copywriter-box').classList.remove('hidden');
      }

      function clearAICopywriterOutput() {
        document.getElementById('ai-copywriter-box').classList.add('hidden');
      }

      /* AI IMPLEMENTATION ENGINE BLOCK 2: DISPATCH MESSAGE CONTEXT MAKER */
      function copyAIDispatchTextToClipboard(title, handle) {
        const cleanHandle = handle.startsWith('@') ? handle : `@${handle}`;

        const dispatchBlueprints = [
          `Hey ${cleanHandle}! Quick update from the ThriftTrack hangar 🚀 Your payment for the "${title}" has been verified successfully. We are currently wrapping it up securely and it will be dispatched today. Get your fit ready! 📦✨`,
          `Hi ${cleanHandle}! Payment received safely for your "${title}" 🛍️. It is officially marked out of the store inventory. The courier team is printing your delivery airway bill right now. Thank you for choosing sustainable fashion! 🌎🔥`,
          `Confirmed, ${cleanHandle}! 💵 Your order for the "${title}" is validated. Processing packaging metrics immediately so it drops on your doorstep ASAP. Keep an eye out for our dispatch rider!`,
        ];

        const randomMessage =
          dispatchBlueprints[
            Math.floor(Math.random() * dispatchBlueprints.length)
          ];

        const fallbackArea = document.getElementById('clipboard-fallback');
        fallbackArea.value = randomMessage;
        fallbackArea.select();
        fallbackArea.setSelectionRange(0, 99999);

        navigator.clipboard
          .writeText(randomMessage)
          .then(() => {
            alert(
              `AI Dispatch Text Copied For ${cleanHandle}!\n\nYou can now go directly to Instagram/WhatsApp and paste it to the customer.`
            );
          })
          .catch(() => {
            document.execCommand('copy');
            alert(`AI Dispatch Text Copied For ${cleanHandle}!`);
          });
      }

      function copyToClipboard(elementId, successAlertText) {
        const text = document.getElementById(elementId).innerText;
        navigator.clipboard.writeText(text).then(() => {
          alert(successAlertText);
        });
      }

      async function executeCreateThriftDrop() {
        const titleInput = document.getElementById('drop-title-input');
        const priceInput = document.getElementById('drop-price-input');
        const title = titleInput.value.trim();
        const price = Number(priceInput.value.trim());

        if (!title || !price) {
          alert(
            'Please fill in both the Drop Item Title and the Listed Price value.'
          );
          return;
        }

        if (adminSupabase) {
          const { error } = await adminSupabase
            .from('items')
            .insert([
              {
                title: title,
                price: price,
                status: 'available',
                is_paid: false,
              },
            ]);

          if (error) {
            alert(`Drop upload failure: ${error.message}`);
          } else {
            alert(`Success! "${title}" has been added to the stock room.`);
            titleInput.value = '';
            priceInput.value = '';
            clearAICopywriterOutput();
            switchTab('dashboard');
            loadWorkspacePipeline();
          }
        }
      }

      async function loadWorkspacePipeline() {
        if (!adminSupabase) return;
        try {
          const { data: allItems, error } = await adminSupabase
            .from('items')
            .select('*')
            .order('created_at', { ascending: false });
          if (error) throw error;

          const availableItems = allItems.filter(
            (i) => i.status === 'available' || !i.status
          );
          const pendingItems = allItems.filter(
            (i) => i.status === 'sold' && !i.is_paid
          );
          const settledItems = allItems.filter((i) => i.is_paid === true);
          const dailyRevenue = settledItems.reduce(
            (acc, current) => acc + Number(current.price || 0),
            0
          );

          document.getElementById('metric-active').innerText =
            availableItems.length;
          document.getElementById('metric-pending').innerText =
            pendingItems.length;
          document.getElementById(
            'metric-sales'
          ).innerText = `₦${dailyRevenue.toLocaleString()}`;
          document.getElementById(
            'inventory-count-tag'
          ).innerText = `${allItems.length} Records`;

          const selectMenu = document.getElementById('quick-lock-item');
          selectMenu.innerHTML =
            '<option value="">-- Choose active item --</option>';
          availableItems.forEach((item) => {
            selectMenu.innerHTML += `<option value="${item.id}">${
              item.title
            } - ₦${Number(item.price).toLocaleString()}</option>`;
          });

          const feedContainer = document.getElementById(
            'ledger-feed-container'
          );
          if (allItems.length === 0) {
            feedContainer.innerHTML = `<p class="p-5 text-slate-400 text-center">No transaction records present.</p>`;
          } else {
            feedContainer.innerHTML = '';
            allItems.forEach((item) => {
              let badgeClass = 'bg-slate-100 text-slate-700';
              let statusLabel = 'Available Floor';

              if (item.status === 'sold' && !item.is_paid) {
                badgeClass = 'bg-amber-100 text-amber-800 animate-pulse';
                statusLabel = 'Awaiting Payout Verification';
              } else if (item.is_paid) {
                badgeClass = 'bg-emerald-100 text-emerald-800';
                statusLabel = 'Settled & Dispatched';
              }

              feedContainer.innerHTML += `
                <div class="p-4 flex flex-col sm:flex-row sm:items-center justify-between gap-2 hover:bg-slate-50 transition">
                  <div>
                    <div class="flex items-center gap-2 flex-wrap">
                      <span class="font-bold text-slate-900 text-sm">${
                        item.title
                      }</span>
                      <span class="px-2 py-0.5 rounded text-[10px] font-bold ${badgeClass}">${statusLabel}</span>
                    </div>
                    <p class="text-slate-400 text-[11px] mt-0.5">
                      ${
                        item.buyer_handle
                          ? `Customer: <strong>${item.buyer_handle}</strong>`
                          : 'Unclaimed Public Stock Card'
                      } 
                      ${item.channel ? ` • via ${item.channel}` : ''}
                    </p>
                  </div>
                  <div class="flex items-center gap-3 justify-between sm:justify-end border-t sm:border-t-0 pt-2 sm:pt-0">
                    <span class="font-bold text-slate-900 text-sm mr-2">₦${Number(
                      item.price
                    ).toLocaleString()}</span>
                    
                    <div class="flex items-center gap-2">
                      ${
                        item.is_paid && item.buyer_handle
                          ? `
                        <button onclick="copyAIDispatchTextToClipboard('${item.title.replace(
                          /'/g,
                          "\\'"
                        )}', '${item.buyer_handle.replace(
                              /'/g,
                              "\\'"
                            )}')" class="bg-slate-900 hover:bg-slate-800 text-white text-[11px] font-bold px-2.5 py-1.5 rounded-lg transition shadow-sm flex items-center gap-1">
                          <span>💬 DM Dispatch Text</span>
                        </button>
                      `
                          : ''
                      }

                      ${
                        item.status === 'sold' && !item.is_paid
                          ? `
                        <button onclick="approveLedgerPayout('${item.id}')" class="bg-emerald-600 hover:bg-emerald-700 text-white text-[11px] font-bold px-2.5 py-1.5 rounded-lg transition shadow-sm">
                          Confirm Funds Receipt
                        </button>
                      `
                          : ''
                      }
                    </div>
                  </div>
                </div>`;
            });
          }

          // Render Master Inventory Registry Data Matrix Table
          const tableBody = document.getElementById('inventory-table-body');
          if (allItems.length === 0) {
            tableBody.innerHTML = `<tr><td colspan="4" class="p-5 text-slate-400 text-center">No active stock logs registered.</td></tr>`;
          } else {
            tableBody.innerHTML = '';
            allItems.forEach((item) => {
              const isEditing = item.id === editingItemId;

              if (isEditing) {
                tableBody.innerHTML += `
                  <tr class="bg-indigo-50/50 border-y border-indigo-100">
                    <td class="p-3">
                      <input type="text" id="inline-edit-title-${item.id}" value="${item.title}" class="w-full text-xs font-bold text-slate-900 bg-white border border-slate-200 rounded-lg px-2.5 py-1.5 focus:outline-none focus:border-indigo-500" />
                    </td>
                    <td class="p-3">
                      <input type="number" id="inline-edit-price-${item.id}" value="${item.price}" class="w-28 text-xs font-mono font-bold text-slate-900 bg-white border border-slate-200 rounded-lg px-2.5 py-1.5 focus:outline-none focus:border-indigo-500" />
                    </td>
                    <td class="p-3">
                      <span class="text-[10px] uppercase font-bold text-slate-400 tracking-wider">Editing Mode</span>
                    </td>
                    <td class="p-3 text-right space-x-2 whitespace-nowrap">
                      <button onclick="executeSaveInlineEdit('${item.id}')" class="bg-emerald-600 hover:bg-emerald-700 text-white text-[11px] font-bold px-3 py-1.5 rounded-lg shadow-sm transition">
                        Save
                      </button>
                      <button onclick="toggleInlineEditMode(null)" class="bg-slate-200 hover:bg-slate-300 text-slate-700 text-[11px] font-semibold px-3 py-1.5 rounded-lg transition">
                        Cancel
                      </button>
                    </td>
                  </tr>`;
              } else {
                tableBody.innerHTML += `
                  <tr class="hover:bg-slate-50/80 transition border-b border-slate-100">
                    <td class="p-4 font-bold text-slate-900">${item.title}</td>
                    <td class="p-4 font-mono font-semibold">₦${Number(
                      item.price
                    ).toLocaleString()}</td>
                    <td class="p-4">
                      <span class="px-2 py-0.5 rounded text-[10px] font-black uppercase tracking-wider ${
                        item.is_paid
                          ? 'bg-emerald-50 text-emerald-700'
                          : item.status === 'sold'
                          ? 'bg-amber-50 text-amber-700'
                          : 'bg-slate-100 text-slate-600'
                      }">
                        ${
                          item.is_paid
                            ? 'Settled'
                            : item.status === 'sold'
                            ? 'Awaiting Pay'
                            : 'On Floor'
                        }
                      </span>
                    </td>
                    <td class="p-4 text-right space-x-3 whitespace-nowrap">
                      <button onclick="toggleInlineEditMode('${
                        item.id
                      }')" class="text-indigo-600 hover:text-indigo-800 font-bold hover:underline transition">
                        Edit Details
                      </button>
                      <button onclick="deleteInventoryItem('${
                        item.id
                      }')" class="text-rose-500 hover:text-rose-700 font-bold hover:underline transition">
                        Delete Record
                      </button>
                    </td>
                  </tr>`;
              }
            });
          }
        } catch (err) {
          console.error('Pipeline query compilation error:', err);
        }
      }

      function toggleInlineEditMode(itemId) {
        editingItemId = itemId;
        loadWorkspacePipeline();
      }

      async function executeSaveInlineEdit(itemId) {
        if (!adminSupabase) return;

        const updatedTitle = document
          .getElementById(`inline-edit-title-${itemId}`)
          .value.trim();
        const updatedPrice = Number(
          document.getElementById(`inline-edit-price-${itemId}`).value.trim()
        );

        if (!updatedTitle || !updatedPrice) {
          alert(
            'Item identity names and listed price points cannot be saved empty.'
          );
          return;
        }

        const { error } = await adminSupabase
          .from('items')
          .update({ title: updatedTitle, price: updatedPrice })
          .eq('id', itemId);

        if (error) {
          alert(`Failed to save adjustments: ${error.message}`);
        } else {
          editingItemId = null;
          loadWorkspacePipeline();
        }
      }

      async function executeManualQuickLock() {
        const itemId = document.getElementById('quick-lock-item').value;
        const buyerHandle = document
          .getElementById('quick-lock-handle')
          .value.trim();
        const originChannel =
          document.getElementById('quick-lock-channel').value;

        if (!itemId || !buyerHandle) {
          alert(
            'Please select a catalog item and assign a buyer profile identifier handle.'
          );
          return;
        }

        if (adminSupabase) {
          const { error } = await adminSupabase
            .from('items')
            .update({
              status: 'sold',
              buyer_handle: buyerHandle,
              channel: originChannel,
              is_paid: false,
            })
            .eq('id', itemId);

          if (!error) {
            document.getElementById('quick-lock-handle').value = '';
            loadWorkspacePipeline();
          }
        }
      }

      async function approveLedgerPayout(itemId) {
        if (adminSupabase) {
          const { error } = await adminSupabase
            .from('items')
            .update({ is_paid: true })
            .eq('id', itemId);
          if (!error) {
            alert(
              'Payment ledger verified! Balance moved into gross revenue metrics.'
            );
            loadWorkspacePipeline();
          }
        }
      }

      async function deleteInventoryItem(itemId) {
        if (
          !confirm(
            'Are you certain you want to strike this item record out of persistent catalog tables permanently?'
          )
        )
          return;
        if (adminSupabase) {
          const { error } = await adminSupabase
            .from('items')
            .delete()
            .eq('id', itemId);
          if (!error) loadWorkspacePipeline();
        }
      }

      function savePayoutCoordinates() {
        if (appRoleState !== 'owner') {
          alert(
            'Access Violation: Assistant context keys lack writing authorization flags over configurations.'
          );
          return;
        }
        alert(
          'Success! Coordinates pushed cleanly into checkout window models.'
        );
      }
